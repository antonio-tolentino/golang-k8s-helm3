## Example of secret with IAM service account json key.
## Service account must have Storage Admin role
 
## Rename json service account file to "jenkins-gcp-credentials.json" and create secret
kubectl create secret generic kaniko-gcp-credentials-secret --from-file=./jenkins-gcp-credentials.json



## Example for configmap with docker hub credentials

kubectl create configmap docker-config --from-file=config.json
